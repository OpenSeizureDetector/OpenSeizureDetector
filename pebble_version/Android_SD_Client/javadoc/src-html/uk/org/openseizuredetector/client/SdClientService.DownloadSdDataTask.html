159    /**
160     * onStartCommand - start regular download of status data from the server.
161     */
162    @Override
163    public int onStartCommand(Intent intent, int flags, int startId) {
164        Log.v(TAG,"SdClientService service starting");
165        
166        // Update preferences.
167        Log.v(TAG,"calling updatePrefs()");
168        updatePrefs();
169        
170        // Display a notification icon in the status bar of the phone to
171        // show the service is running.
172        Log.v(TAG,"showing Notification");
173        showNotification();
174
175        // Start timer to check status of pebble regularly.
176        mStatusTime = new Time(Time.getCurrentTimezone());
177        mStatusTime.setToNow();
178        if (statusTimer==null) {
179            Log.v(TAG,"onCreate(): starting status timer");
180            statusTimer = new Timer();
181            statusTimer.schedule(new TimerTask() {
182                    @Override
183                    public void run() {getSdData();}
184                }, 0, mDataUpdatePeriod);       
185        } else {
186            Log.v(TAG,"onCreate(): status timer already running.");
187        }
188        
189        // Apply the wake-lock to prevent CPU sleeping (very battery intensive!)
190        if (mWakeLock!=null) {
191            mWakeLock.acquire();
192            Log.v(TAG,"Applied Wake Lock to prevent device sleeping");
193        } else {
194            Log.d(TAG,"mmm...mWakeLock is null, so not aquiring lock.  This shouldn't happen!");
195        }
196
197        return START_STICKY;
198    }
199
200    @Override
201    public void onDestroy() {
202        Log.v(TAG,"onDestroy(): Service stopping");
203        // release the wake lock to allow CPU to sleep and reduce
204        // battery drain.
205        if (mWakeLock!=null) {
206            mWakeLock.release();
207            Log.v(TAG,"Released Wake Lock to allow device to sleep.");
208        } else {
209            Log.d(TAG,"mmm...mWakeLock is null, so not releasing lock.  This shouldn't happen!");
210        }
211
212        try {
213            // Stop the status timer
214            if (statusTimer!=null) {
215                Log.v(TAG,"onDestroy(): cancelling status timer");
216                statusTimer.cancel();
217                statusTimer.purge();
218                statusTimer = null;
219            }
220
221            // Cancel the notification.
222            Log.v(TAG,"onDestroy(): cancelling notification");
223            mNM.cancel(NOTIFICATION_ID);
224
225            // stop this service.
226            Log.v(TAG,"onDestroy(): calling stopSelf()");
227            stopSelf();
228
229        } catch(Exception e) {
230            Log.v(TAG,"Error in onDestroy() - "+e.toString());
231        }
232    }

255    /*
256     * Inhibit fault alarm initiation for a period to avoid spurious warning 
257     * beeps caused by short term network interruptions.
258     */
259    private class FaultTimer extends CountDownTimer {
260        public long mFaultTimerRemaining = 0;
261        public FaultTimer(long startTime, long interval) {
262            super(startTime, interval);
263        }       
264        @Override
265        public void onFinish() {
266            mFaultTimerCompleted = true;
267            Log.v(TAG,"mFaultTimer - removing mFaultTimerRunning flag");
268        }
269        @Override
270        public void onTick(long msRemaining) {
271            mFaultTimerRemaining = msRemaining/1000;
272            Log.v(TAG,"mFaultTimer - onTick() - Time Remaining = "
273                  + mFaultTimerRemaining);
274        }
275        
276    }

301    /** 
302     * Start the fault timer that is used to require a fault to remain
303     * standing for a period before raising fault beeps.
304     */
305    public void startFaultTimer() {
306        if (mFaultTimer!=null) {
307            Log.v(TAG,"startFaultTimer(): fault timer already running - not doing anything.");
308        } else {
309            Log.v(TAG,"startFaultTimer(): starting fault timer.");
310            mFaultTimerCompleted = false;
311            mFaultTimer = 
312                // conver to ms.
313                new FaultTimer(mFaultTimerPeriod*1000,1000);
314            mFaultTimer.start();
315        }
316    }
317
318    public void stopFaultTimer() {
319        if (mFaultTimer!=null) {
320            Log.v(TAG,"stopFaultTimer(): fault timer already running - cancelling it.");
321            mFaultTimer.cancel();
322            mFaultTimer = null;
323            mFaultTimerCompleted = false;
324        } else {
325            Log.v(TAG,"stopFaultTimer(): fault timer not running - not doing anything.");
326        }
327    }

330    public void cancelAudible() {
331        // Start timer to remove the cancel audible flag
332        // after the required period.
333        if (mCancelAudibleTimer!=null) {
334            Log.v(TAG,"onCreate(): cancel audible timer already running - cancelling it.");
335            mCancelAudibleTimer.cancel();
336            mCancelAudibleTimer = null;
337            mCancelAudible = false;
338        } else {
339            Log.v(TAG,"cancelAudible(): starting cancel audible timer");
340            mCancelAudible = true;
341            mCancelAudibleTimer = 
342                // conver to ms.
343                new CancelAudibleTimer(mCancelAudiblePeriod*60*1000,1000);
344            mCancelAudibleTimer.start();
345        }
346    }

379                }

381        } else {

383            Log.v(TAG,"faultWarningBeep() - starting Fault Timer");
384        }

472    /**
473     * Check the seizure detector data for alarm conditions and beep and
474     * display the ClientActivity if an alarm condition is shown.
475     */
476    private void checkAlarms() {
477        Log.v(TAG,"checkAlarms()");
478            // Check the alarm state, and raise alarms if necessary.
479            if (mSdData.alarmState==0) {
480                Log.v(TAG,"Status=OK");
481            }
482            if (mSdData.alarmState==1) {
483                Log.v(TAG,"Status=WARNING");
484                warningBeep();
485            }
486            if (mSdData.alarmState==2) {
487                Log.v(TAG,"Status=ALARM");
488                alarmBeep();
489                Intent clientActivityIntent = 
490                    new Intent(this, ClientActivity.class);
491                clientActivityIntent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
492                startActivity(clientActivityIntent);
493            }
494            if ((mSdData.pebbleConnected == false) || 
495                (mSdData.pebbleAppRunning == false)) {
496                faultWarningBeep();
497            } else {
498                stopFaultTimer();
499            }
500    }

504    /**
505     * Retrive the current Seizure Detector Data from the server.
506     * Uses teh DownloadSdDataTask class to download the data in the
507     * background.  The data is processed in DownloadSdDataTask.onPostExecute().
508     */
509    private void getSdData() {
510        Log.v(TAG,"getSdData()");
511        ConnectivityManager connMgr = (ConnectivityManager) 
512            getSystemService(Context.CONNECTIVITY_SERVICE);
513        NetworkInfo networkInfo = connMgr.getActiveNetworkInfo();
514        //if (networkInfo != null && networkInfo.isConnected()) {
515        if (true) {
516            new DownloadSdDataTask().execute("http://"+mServerIP+":8080/data");
517        } else {
518            Log.v(TAG,"No network connection available.");
519        }    }
520    
521    private class DownloadSdDataTask extends AsyncTask<String, Void, String> {
522        @Override
523            protected String doInBackground(String... urls) {
524            // params comes from the execute() call: params[0] is the url.
525            try {
526                return downloadUrl(urls[0]);
527            } catch (IOException e) {
528                return "Unable to retrieve web page. URL may be invalid.";
529            }
530        }
531        // onPostExecute displays the results of the AsyncTask.
532        @Override
533            protected void onPostExecute(String result) {
534            //Log.v(TAG,"onPostExecute() - result="+result);
535            if (result.startsWith("Unable to retrieve web page")) {
536                    Log.v(TAG,"onPostExecute - Unable to retrieve data");
537                    mSdData.serverOK = false;
538                    mSdData.pebbleConnected = false;
539                    mSdData.pebbleAppRunning = false;
540                    mSdData.alarmState = 0;
541                    mSdData.alarmPhrase = "Warning - No Connection to Server";
542                    //faultWarningBeep();
543                } else {
544                    // Populate mSdData using the received data.
545                    mSdData.serverOK = true;
546                    mStatusTime.setToNow();
547                    mSdData.fromJSON(result);
548                    Log.v(TAG,"onPostExecute(): mSdData = "+mSdData.toString());
549                }
550            checkAlarms();
551        }
552    }

554    // Given a URL, establishes an HttpUrlConnection and retrieves
555    // the web page content as a InputStream, which it returns as
556    // a string.
557    private String downloadUrl(String myurl) throws IOException {
558        InputStream is = null;
559        // Only display the first 500 characters of the retrieved
560        // web page content.
561        int len = 500;
562        
563        try {
564            URL url = new URL(myurl);
565            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
566            conn.setReadTimeout(5000 /* milliseconds */);
567            conn.setConnectTimeout(5000 /* milliseconds */);
568            conn.setRequestMethod("GET");
569            conn.setDoInput(true);
570            // Starts the query
571            conn.connect();
572            int response = conn.getResponseCode();
573            Log.d(TAG, "The response is: " + response);
574            is = conn.getInputStream();
575            
576            // Convert the InputStream into a string
577            String contentAsString = readInputStream(is, len);
578            return contentAsString;
579            
580            // Makes sure that the InputStream is closed after the app is
581            // finished using it.
582        } finally {
583            if (is != null) {
584                is.close();
585            } 
586        }
587    }

589    // Reads an InputStream and converts it to a String.
590    public String readInputStream(InputStream stream, int len) throws IOException, UnsupportedEncodingException {
591        Reader reader = null;
592        reader = new InputStreamReader(stream, "UTF-8");        
593        char[] buffer = new char[len];
594        reader.read(buffer);
595        return new String(buffer);
596    }
