171    @Override
172    public boolean onOptionsItemSelected(MenuItem item) {
173        Log.v(TAG,"Option "+item.getItemId()+" selected");
174        switch (item.getItemId()) {
175        case R.id.action_start_stop:
176            // Respond to the start/stop server menu item.
177            Log.v(TAG,"action_sart_stop");
178            if (mBound) {
179                Log.v(TAG,"Stopping Server");
180                stopServer();
181            } else {
182                Log.v(TAG,"Starting Server");
183                startServer();
184            }
185            return true;
186        case R.id.action_test_fault_warning_beep:
187            Log.v(TAG,"action_test_fault_warning_beep");
188            if (mBound) {
189                mSdClientService.faultWarningBeep();
190            }
191            return true;
192        case R.id.action_test_alarm_beep:
193            Log.v(TAG,"action_test_alarm_beep");
194            if (mBound) {
195                mSdClientService.alarmBeep();
196            }
197            return true;
198        case R.id.action_test_warning_beep:
199            Log.v(TAG,"action_test_warning_beep");
200            if (mBound) {
201                mSdClientService.warningBeep();
202            }
203            return true;
204        case R.id.action_settings:
205            Log.v(TAG,"action_settings");
206            try {
207                Intent prefsIntent = new Intent(
208                                                ClientActivity.this,
209                                                PrefActivity.class);
210                this.startActivity(prefsIntent);
211            } catch (Exception ex) {
212                Log.v(TAG,"exception starting settings activity "+ex.toString());
213            }
214            return true;
215        default:
216            return super.onOptionsItemSelected(item);
217        }
218    }

220    @Override
221    protected void onStart() {
222        super.onStart();
223        SharedPreferences SP = PreferenceManager
224           .getDefaultSharedPreferences(getBaseContext());
225
226        try {
227            String uiUpdatePeriodStr = SP.getString("UiUpdatePeriod","2000");
228            mUiUpdatePeriod = Integer.parseInt(uiUpdatePeriodStr);
229            Log.v(TAG,"onStart() - mUiUpdatePeriod = "+mUiUpdatePeriod);
230        } catch (Exception ex) {
231            Log.v(TAG,"onStart() - Problem parsing preferences!");
232            Toast toast = Toast.makeText(getApplicationContext(),"Problem Parsing Preferences - Something won't work",Toast.LENGTH_SHORT);
233            toast.show();
234        }
235
236        // start timer to refresh user interface every 5 seconds.
237        if (mUiTimer==null) {
238            Log.v(TAG,"onstart(): starting mUiTimer.");
239            mUiTimer = new Timer();
240            mUiTimer.schedule(new TimerTask() {
241                    @Override
242                    public void run() {updateUI();}
243                }, 0, mUiUpdatePeriod); 
244            Log.v(TAG,"onStart(): started mUiTimer");
245        } else {
246            Log.v(TAG,"onStart(): mUiTimer already running");
247        }
248
249
250        TextView tv;
251        tv = (TextView) findViewById(R.id.versionTv);
252        String versionName = "unknown";
253        // From http://stackoverflow.com/questions/4471025/
254        //         how-can-you-get-the-manifest-version-number-
255        //         from-the-apps-layout-xml-variable
256        final PackageManager packageManager = getPackageManager();
257        if (packageManager != null) {
258            try {
259                PackageInfo packageInfo = packageManager.getPackageInfo(getPackageName(), 0);
260                versionName = packageInfo.versionName;
261            } catch (PackageManager.NameNotFoundException e) {
262                Log.v(TAG,"failed to find versionName");
263                versionName = null;
264            }
265        }
266        tv.setText("OpenSeizureDetector Client Version "+versionName);
267
268        startServer();
269
270    }

272    @Override

274        super.onStop();
275        // Disconnect from background service.
276        if (mBound) {

278            mBound = false;
279        }

281            // Stop the status timer

284                mUiTimer.cancel();
285                mUiTimer.purge();
286                mUiTimer = null;
287            }

292  /** Defines callbacks for service binding, passed to bindService() */
293    private ServiceConnection mConnection = new ServiceConnection() {
294
295        @Override
296        public void onServiceConnected(ComponentName className,
297                IBinder service) {
298            // We've bound to LocalService, cast the IBinder and get LocalService instance
299            SdClientService.SdBinder binder = (SdClientService.SdBinder) service;
300            mSdClientService = binder.getService();
301            mBound = true;
302            if (mSdClientService!=null) {
303                Log.v(TAG,"onServiceConnected() - Asking server to update its settings");
304                mSdClientService.updatePrefs();
305            }
306            else {
307                Log.v(TAG,"onServiceConnected() - mSdClientService is null - this is wrong!");
308            }
309        }
310
311        @Override
312        public void onServiceDisconnected(ComponentName arg0) {
313            Log.v(TAG,"onServiceDisonnected()");
314            mBound = false;
315        }
316    };

318    /**
319     * Start the SdClientService service
320     */
321    private void startServer() {
322        // Start the server
323        sdClientServiceIntent = new Intent(ClientActivity.this,SdClientService.class);
324        sdClientServiceIntent.setData(Uri.parse("Start"));
325        getApplicationContext().startService(sdClientServiceIntent);
326
327        // and bind to it so we can see its data
328        Intent intent = new Intent(this,SdClientService.class);
329        bindService(intent,mConnection, Context.BIND_AUTO_CREATE);
330        
331        // Change the action bar icon to show the option to stop the service.
332        if (mOptionsMenu!=null) {
333            Log.v(TAG,"Changing menu icons");
334            MenuItem menuItem = mOptionsMenu.findItem(R.id.action_start_stop);
335            menuItem.setIcon(R.drawable.stop_server);
336            menuItem.setTitle("Stop Server");
337        } else {
338            Log.v(TAG,"mOptionsMenu is null - not changing icons!");
339        }
340    }
341
342    /**
343     * Stop the SdClientService service
344     */
345    private void stopServer() {
346        Log.v(TAG,"stopping Server...");
347        // unbind this activity from the service if it is bound.
348        if (mBound) {
349            try {
350                unbindService(mConnection);
351                mBound = false;
352            } catch (Exception ex) {
353                Log.e(TAG,"stopServer - error unbinding service - "+ex.toString());
354            }
355        }
356        // then send an Intent to stop the service.
357        sdClientServiceIntent = new Intent(ClientActivity.this,SdClientService.class);
358        sdClientServiceIntent.setData(Uri.parse("Stop"));
359        getApplicationContext().stopService(sdClientServiceIntent);
360
361        // Change the action bar icon to show the option to start the service.
362        if (mOptionsMenu!=null) {
363            Log.v(TAG,"Changing action bar icons");
364            mOptionsMenu.findItem(R.id.action_start_stop).setIcon(R.drawable.start_server);
365            mOptionsMenu.findItem(R.id.action_start_stop).setTitle("Start Server");
366        } else {
367            Log.v(TAG,"mOptionsMenu is null, not changing icons!");
368        }
369
370    }

382            if ("uk.org.openseizuredetector.SdClientService"

386            }
387        }
388        //Log.v(TAG,"No!");

396    /*
397     * updateUI - called by the uiTimer timer periodically.
398     * requests the ui to be updated by calling serverStatusRunnable.
399     */
400    private void updateUI() {
401        Log.v(TAG,"updateUI()");
402        serverStatusHandler.post(updateUIRunnable);
403    }
404        
405    /*
406     * updateUIRunnable - called by updateUI - updates the
407     * user interface to reflect the current status received from the server.
408     */
409    final Runnable updateUIRunnable = new Runnable() {
410            public void run() {
411                if (mSdClientService!=null) {
412                    TextView tv;
413                    tv = (TextView) findViewById(R.id.textView1);
414                    if (mSdClientService.mSdData.serverOK) {   // was isServerRunning
415                        tv.setText("Server Running OK");
416                        tv.setBackgroundColor(okColour);
417                        tv = (TextView)findViewById(R.id.textView2);
418                        tv.setText("Connecting to Server at http://"
419                                   +mSdClientService.mServerIP+":8080");
420                        tv.setBackgroundColor(okColour);
421                    } else {
422                        tv.setText("***Can Not Access Server***");
423                        tv.setBackgroundColor(alarmColour);
424                    }
425                 
426                    // Deal with Cancel Audible button
427                    Button cancelAudibleButton = 
428                        (Button) findViewById(R.id.cancelAudibleButton);
429                    if (mSdClientService.isAudibleCancelled()) {
430                        cancelAudibleButton.setText("Audible Alarms Cancelled "
431                                                    + "for "
432                                                    + mSdClientService.
433                                                    cancelAudibleTimeRemaining()
434                                                    + " sec."
435                                                    + " Press to re-enable");
436                    } else {
437                        if (mSdClientService.mAudibleAlarm) {
438                            cancelAudibleButton.setText("Cancel Audible Alarms (temporarily)");
439                        } else {
440                            cancelAudibleButton.setText("Audible Alarms OFF");
441                        }
442                    }
443                    try {
444                        tv = (TextView) findViewById(R.id.alarmTv);
445                        if (mSdClientService.mSdData.alarmState==0) {
446                            tv.setText(mSdClientService.mSdData.alarmPhrase);
447                            tv.setBackgroundColor(okColour);
448                        }
449                        if (mSdClientService.mSdData.alarmState==1) {
450                            tv.setText(mSdClientService.mSdData.alarmPhrase);
451                            tv.setBackgroundColor(warnColour);
452                        }
453                        if (mSdClientService.mSdData.alarmState==2) {
454                            tv.setText(mSdClientService.mSdData.alarmPhrase);
455                            tv.setBackgroundColor(alarmColour);
456                        }
457                        tv = (TextView) findViewById(R.id.pebTimeTv);
458                        tv.setText(mSdClientService.mSdData.dataTime.format("%H:%M:%S"));
459                        // Pebble Connected Phrase
460                        tv = (TextView) findViewById(R.id.pebbleTv);
461                        if (mSdClientService.mSdData.pebbleConnected) {
462                            tv.setText("Pebble Watch Connected OK");        
463                            tv.setBackgroundColor(okColour);
464                        } else {
465                            tv.setText("** Pebble Watch NOT Connected **");
466                            tv.setBackgroundColor(alarmColour);
467                        }
468                        tv = (TextView) findViewById(R.id.appTv);
469                        if (mSdClientService.mSdData.pebbleAppRunning) {
470                            tv.setText("Pebble App OK");            
471                        tv.setBackgroundColor(okColour);
472                        } else {
473                            tv.setText("** Pebble App NOT Running **");     
474                            tv.setBackgroundColor(alarmColour);
475                        }
476                        tv = (TextView) findViewById(R.id.battTv);
477                        tv.setText("Pebble Battery = "+String.valueOf(mSdClientService.mSdData.batteryPc)+"%");
478                        if (mSdClientService.mSdData.batteryPc<=20)
479                            tv.setBackgroundColor(alarmColour);
480                        if (mSdClientService.mSdData.batteryPc>20)
481                            tv.setBackgroundColor(warnColour);
482                        if (mSdClientService.mSdData.batteryPc>=40)
483                            tv.setBackgroundColor(okColour);
484                        
485                        tv = (TextView) findViewById(R.id.debugTv);
486                        String specStr = "";
487                        for (int i=0;i<10;i++)
488                            specStr = specStr 
489                                + mSdClientService.mSdData.simpleSpec[i] 
490                                + ", ";
491                        tv.setText("Spec = "+specStr);
492                    } catch (Exception e) {
493                        Log.v(TAG,"updateUIRunnable: Exception - "+e.toString());
494                    }
495                    ////////////////////////////////////////////////////////////
496                    // Produce graph
497                    LineChart mChart = (LineChart) findViewById(R.id.chart1);
498                    mChart.setDescription("");
499                    mChart.setNoDataTextDescription("You need to provide data for the chart.");
500                    // X Values
501                    ArrayList<String> xVals = new ArrayList<String>();
502                    for (int i = 0; i < 10; i++) {
503                        xVals.add((i) + "");
504                    }
505                    // Y Values
506                    ArrayList<Entry> yVals = new ArrayList<Entry>();
507                    for (int i = 0; i < 10; i++) {
508                        if (mSdClientService.mSdData!=null)
509                            yVals.add(new Entry(mSdClientService.mSdData.simpleSpec[i], i));
510                        else
511                            yVals.add(new Entry(i, i));
512                    }
513                    
514                    // create a dataset and give it a type
515                    LineDataSet set1 = new LineDataSet(yVals, "DataSet 1");
516                    set1.setColor(Color.BLACK);
517                    set1.setLineWidth(1f);
518                    
519                    ArrayList<LineDataSet> dataSets = new ArrayList<LineDataSet>();
520                    dataSets.add(set1); // add the datasets
521                    LineData data = new LineData(xVals, dataSets);
522                    //data.setValueTextSize(10f);
523                    mChart.setData(data);
524                    mChart.invalidate();
525                } else {
526                    Log.v(TAG,"mSdClientService is null - service not running?");
527                }
528            }
529        };
