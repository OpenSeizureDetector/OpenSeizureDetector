007import java.util.Iterator;
008import java.util.List;
009import android.content.Context;

019interface LocationReceiver {
020    /** The function to be called once we have found the location */
021    public void onLocationFound(LonLat ll);
022}
023
024/**
025 * @author Graham Jones
026 * Based on various examples from the internet.
027 * Call with 
028 *                      LocationFinder lf = new LocationFinder(this);
029 *                      LonLat ll = lf.getLocationLL();
030 *
031 */
032public class LocationFinder implements LocationListener, Runnable, GpsStatus.Listener {
033    LocationManager locMgr;
034    LocationReceiver lr;
035    String mProvider;
036    long mTimeStart;
037    Context mContext;
038    boolean mTimedOut = false;
039    int timeOutCount = 0;
040    int timeOutSec;
041    Handler mHandler;
042    boolean mUseGPS = true;
043    String TAG = "LocationFinder";
044    
045    public LocationFinder(Context contextArg) {
046        mContext = contextArg;
047        mHandler = new Handler();
048        mUseGPS = false;
049        locMgr = (LocationManager)
050            contextArg.getSystemService(
051                                        Context.LOCATION_SERVICE);
052        
053        if (locMgr==null) {
054            Log.v("LocationFinder","ERROR - locMgr is null - THIS WILL NOT WORK!!!");
055        }
056        
057        // List all providers to log.:
058        List<String> providers = locMgr.getAllProviders();
059        for (String provider : providers) {
060            Log.v("LocationFinder",provider+","+locMgr.getProvider(provider).toString());
061        }
062        
063    }
064    
065    
066    public void getLocationLL(LocationReceiver lr, int timeOutSec, boolean useGPS) {
067        this.timeOutSec = timeOutSec;
068        mUseGPS = useGPS;
069        if (mUseGPS) {
070            mProvider = LocationManager.GPS_PROVIDER;
071        } else {
072            mProvider = LocationManager.NETWORK_PROVIDER;
073        }
074        Log.v("mProvider",mProvider);
075        this.lr = lr;
076        // Ask for location updates to be sent to the onLocationChanged() method of this class.
077        locMgr.requestLocationUpdates(mProvider, 0,0, this);
078        // Monitor GPS Status while we are searching for a fix.  this.onGpsStatusChanged()
079        // is called when the GPS Status changes.
080        locMgr.addGpsStatusListener(this);
081        
082        // Set a timer running to allow us to give up on getting a GPS fix.  
083        mHandler.removeCallbacks(this);
084        // this.run() is called when the timer times out.
085        mHandler.postDelayed(this, timeOutSec*1000);
086    }
087    
088    public void startFixSearch() {
089        locMgr.requestLocationUpdates(mProvider, 36000,10000, this);            
090    }
091    
092    
093    public void onLocationChanged(Location loc) {
094        Log.v(TAG,"onLocationChanged");
095        if (loc!=null) {
096            Log.v(TAG,"onLocationChanged - mTimedOut ="
097                  +mTimedOut+" Provider="+loc.getProvider());
098            if ((loc.getProvider().equals(mProvider)) || mTimedOut) {
099                locMgr.removeUpdates(this);
100                locMgr.removeGpsStatusListener(this);
101                // Avoid being called again by the timeOut run() function
102                mHandler.removeCallbacks(this); 
103                LonLat ll;
104                ll = new LonLat(loc.getLongitude(),
105                                loc.getLatitude(),
106                                loc.getAccuracy(),
107                                loc.getProvider());
108                lr.onLocationFound(ll);
109            } else {
110                Log.v(TAG,"Skipping location update by "+loc.getProvider()+" mProvider="+mProvider+".");
111                }
112        } else {
113            Log.v(TAG,"loc==null - waiting...");
114        }
115    }
116    
117    public void onProviderDisabled(String provider) {
118        // re-register for updates
119        Log.v("locationListener","onProviderDisabled");
120    }
121    
122    public void onProviderEnabled(String provider) {
123        // is provider better than mProvider?
124        // is yes, mProvider = provider
125        Log.v("locationListener","onProviderEnabled");
126    }
127    
128    public void onStatusChanged(String provider, int status, Bundle extras) {
129        Log.v("locationListener","onStatusChanged");
130    }
131    
132    // Called by the mHandler timer to signify timeout.  
133    // In which case we give up on GPS and fall back onto NETWORK_PROVIDER.
134    // At the moment, if we time out on NETWORK_PROVIDER too, it just
135    // keeps trying over and over - maybe set a count in future to 
136    // make it give up altogether eventually?
137    public void run() {
138        Log.v("WAYN","timeout runnable.run");
139        mTimedOut = true;
140        timeOutCount ++;
141        Log.v(TAG,"TimedOut Number "+timeOutCount+
142              "! - using network location instead.");
143        //Location loc = locMgr.getLastKnownLocation(LocationManager.NETWORK_PROVIDER);
144        //onLocationChanged(loc);
145        
146        // switch off GPS monitoring to save battery.
147        locMgr.removeGpsStatusListener(this);
148        locMgr.removeUpdates(this);
149        
150        // Re-start search using network rather than GPS.
151        mProvider = LocationManager.NETWORK_PROVIDER;
152        Log.v("mProvider",mProvider);
153        // Ask for location updates to be sent to the onLocationChanged() method of this class.
154        locMgr.requestLocationUpdates(mProvider, 0,0, this);
155        
156        // Set a timer running to allow us to give up on getting a fix.  
157        // this.run() is called when the timer times out.
158        mHandler.removeCallbacks(this);
159        mHandler.postDelayed(this, timeOutSec*1000);
160    }
161    
162    public void onGpsStatusChanged(int eventNo) {
163        //if (eventNo == GpsStatus.GPS_EVENT_SATELLITE_STATUS) 
164        {
165            int nSat = 0;
166            GpsStatus gpsStatus = locMgr.getGpsStatus(null);
167            Iterable<GpsSatellite> sats = gpsStatus.getSatellites();
168            Iterator<GpsSatellite> it = sats.iterator();
169            String msg = "Satellites...";
170            while ( it.hasNext() ) {
171                nSat++;
172                GpsSatellite oSat = (GpsSatellite) it.next() ; 
173                Log.v("TEST","LocationActivity - onGpsStatusChange: Satellites: \n" + 
174                      oSat.getSnr() ) ; 
175                msg = msg + oSat.getPrn()+ ", "
176                    + oSat.getSnr()+", "
177                    + oSat.hasAlmanac()+","
178                    + oSat.hasEphemeris()+", \n";
179            } 
180            msg = msg+" - "+nSat+" in view.";
181            Log.v(TAG,msg);
182        }
183    }
