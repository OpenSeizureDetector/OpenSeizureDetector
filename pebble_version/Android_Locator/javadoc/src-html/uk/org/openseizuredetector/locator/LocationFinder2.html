041/**
042 * @author Graham Jones
043 * Based on various examples from the internet.
044 * Call with 
045 *                      LocationFinder lf = new LocationFinder(this);
046 *                      LonLat ll = lf.getLocationLL();
047 * where 'this' must implement the LocationReceiver interface.
048 */
049public class LocationFinder2 implements 
050                                 ConnectionCallbacks, 
051                                 OnConnectionFailedListener,
052                                 LocationListener {
053
054    LocationReceiver lr;
055    Context mContext;
056    protected static final String TAG = "LocationFinder";
057    protected GoogleApiClient mGoogleApiClient;
058    protected Location mLastLocation;
059
060    /**
061     * Constructor for LocationFinder2 - needs a context so it can use
062     * Toast notifications.
063     */
064    public LocationFinder2(Context contextArg) {
065        mContext = contextArg;
066        Log.i(TAG, "Creating connection to Google API");
067        mGoogleApiClient = new GoogleApiClient.Builder(mContext)
068            .addConnectionCallbacks(this)
069            .addOnConnectionFailedListener(this)
070            .addApi(LocationServices.API)
071            .build();
072    }
073
074    public void destructor() {
075        Log.i(TAG, "destructor() - disconnecting from Google API.");
076        if (mGoogleApiClient.isConnected()) {
077            mGoogleApiClient.disconnect();
078        }
079    }
080
081    public void getLocationLL(LocationReceiver lr) {
082        Log.i(TAG, "getLocationLL - connecting to Google API.");
083        this.lr = lr;
084        mGoogleApiClient.connect();
085    }
086
087    /**
088     * Runs when a GoogleApiClient object successfully connects.
089     */
090    @Override
091    public void onConnected(Bundle connectionHint) {
092        Log.i(TAG, "onConnected() - connected to Google API.");
093        // set the parameters for our location determination.
094        LocationRequest locRequest;
095        locRequest = LocationRequest.create();
096        locRequest.setPriority(LocationRequest.PRIORITY_HIGH_ACCURACY);
097        // These intervals should not really matter because we stop the 
098        // request after we receive the first result in onLocationChanged()
099        locRequest.setInterval(10*1000); // 10 seconds.
100        locRequest.setFastestInterval(1*1000); // 1 second
101        // Request location updates - on LocationChanged is called each time
102        // we get a new location notification.
103        LocationServices.FusedLocationApi.requestLocationUpdates(
104                                                 mGoogleApiClient,
105                                                 locRequest,
106                                                 this); 
107        Log.i(TAG, "onConnected() - requesting location updates");
108        Toast.makeText(mContext, "Requesting Location....", Toast.LENGTH_LONG).show();
109    }
110
111    @Override
112    public void onConnectionFailed(ConnectionResult result) {
113        Log.i(TAG, "Connection failed: ConnectionResult.getErrorCode() = " + result.getErrorCode());
114    }
115
116
117    @Override
118    public void onConnectionSuspended(int cause) {
119        Log.i(TAG, "Connection suspended");
120        mGoogleApiClient.connect();
121    }
122
123
124    /**
125     * Called each time we receive a new location notification.
126     * - sends the location back to the LocationReceiver that requested the
127     * location.
128     */ 
129    public void onLocationChanged(Location loc) {
130        Log.i(TAG, "onLocationChanged() - location found.");
131        if (loc != null) {
132            Log.i(TAG, "OnLocationChanged() - returning Location ("
133                  +loc.toString()+")"
134                  );
135            Toast.makeText(mContext,"Location is ("
136                           +loc.getLatitude()+","
137                           +loc.getLongitude()+")",
138                           Toast.LENGTH_LONG);
139            LonLat ll;
140            ll = new LonLat(loc.getLongitude(),
141                            loc.getLatitude(),
142                            loc.getAccuracy(),
143                            loc.getProvider());
144            // Now we have a location, unsubscribe from location updates.
145            LocationServices.FusedLocationApi.removeLocationUpdates(
146                                                 mGoogleApiClient,
147                                                 this); 
148            // And retrn the location to the requesting process.
149            lr.onLocationFound(ll);
150        } else {
151            Log.i(TAG, "onLocationChanged() - Location is null????");
152            Toast.makeText(mContext, "No Location Detected", Toast.LENGTH_LONG).show();
153            lr.onLocationFound(null);
154        }
155    }
