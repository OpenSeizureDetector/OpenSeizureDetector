001package uk.org.openseizuredetector.locator;
002
003import java.text.DecimalFormat;
004import java.text.NumberFormat;
005import java.text.SimpleDateFormat;
006import java.util.Date;
007import java.util.Locale;
008
009public class LonLat {
010    public final double _lon;  //longitude in degrees
011    public final double _lat;  // latitude in degrees
012    public final float _acc;  // fix accuracy in metres.
013    public final Date _date;  // date/time of fix.
014    public String _provider;
015    public double lon() {return _lon;};
016    public double lat() {return _lat;};
017    public String date() {
018        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm",Locale.US);
019        return sdf.format(_date);
020    };
021    public LonLat(double llon,double llat, float lacc, String provider, Date ldate) {
022        _lon = llon;
023        _lat = llat;
024        _acc = lacc;
025        _date = ldate;
026        _provider = provider;
027    }
028
029    /**
030     * Returns location as a human readable string 
031     */
032    String toStr() {
033        NumberFormat df = new DecimalFormat("#0.000");
034        return ("lon="+df.format(_lon)+", lat="+df.format(_lat)+":  accuracy="+df.format(_acc)+" m ("+_provider+")");
035    }
036
037    /**
038     * Returns location as a Geo URI (http://wikipedia.org/wiki/Geo_URI).
039     */
040    String toGeoUri() {
041        NumberFormat df = new DecimalFormat("#0.000");
042        return ("geo:"
043                +df.format(_lat)+","+df.format(_lon)
044                +";u="+df.format(_acc)
045                // The ?q parameter does not display a marker in osmand so
046                // don't bother sending it.
047                //+"?q="+df.format(_lat)+","+df.format(_lon)
048
049                );
050    }
051    
052    
053    
054}
