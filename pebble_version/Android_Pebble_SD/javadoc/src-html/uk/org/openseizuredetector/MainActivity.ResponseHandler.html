154    @Override
155    public boolean onOptionsItemSelected(MenuItem item) {
156        Log.v(TAG,"Option "+item.getItemId()+" selected");
157        switch (item.getItemId()) {
158        case R.id.action_launch_pebble_app:
159            Log.v(TAG,"action_launch_pebble_app");
160            try {
161                PackageManager pm = this.getPackageManager();
162                Intent pebbleAppIntent = pm.getLaunchIntentForPackage("com.getpebble.android");
163                this.startActivity(pebbleAppIntent);
164            } catch (Exception ex) {
165                Log.v(TAG,"exception starting pebble App "+ex.toString());
166            }
167            return true;
168        case R.id.action_start_stop:
169            // Respond to the start/stop server menu item.
170            Log.v(TAG,"action_sart_stop");
171            if (mBound) {
172                Log.v(TAG,"Stopping Server");
173                unbindFromServer();
174                stopServer();
175            } else {
176                Log.v(TAG,"Starting Server");
177                startServer();
178                // and bind to it so we can see its data
179                bindToServer();
180            }
181            return true;
182        case R.id.action_test_fault_beep:
183            Log.v(TAG,"action_test_fault_beep");
184            if (mBound) {
185                mSdServer.faultWarningBeep();
186            }
187            return true;
188        case R.id.action_test_alarm_beep:
189            Log.v(TAG,"action_test_alarm_beep");
190            if (mBound) {
191                mSdServer.alarmBeep();
192            }
193            return true;
194        case R.id.action_test_warning_beep:
195            Log.v(TAG,"action_test_warning_beep");
196            if (mBound) {
197                mSdServer.warningBeep();
198            }
199            return true;
200        case R.id.action_test_sms_alarm:
201            Log.v(TAG,"action_test_sms_alarm");
202            if (mBound) {
203                mSdServer.sendSMSAlarm();
204            }
205            return true;
206        case R.id.action_settings:
207            Log.v(TAG,"action_settings");
208            try {
209                Intent prefsIntent = new Intent(
210                                                MainActivity.this,
211                                                PrefActivity.class);
212                this.startActivity(prefsIntent);
213            } catch (Exception ex) {
214                Log.v(TAG,"exception starting settings activity "+ex.toString());
215            }
216            return true;
217        default:
218            return super.onOptionsItemSelected(item);
219        }
220    }

222    @Override
223    protected void onStart() {
224        super.onStart();
225        SharedPreferences SP = PreferenceManager
226            .getDefaultSharedPreferences(getBaseContext());
227        boolean audibleAlarm = SP.getBoolean("AudibleAlarm",true);
228        Log.v(TAG,"onStart - auidbleAlarm = "+audibleAlarm);
229
230        TextView tv;
231        tv = (TextView) findViewById(R.id.versionTv);
232        String versionName = "unknown";
233        // From http://stackoverflow.com/questions/4471025/
234        //         how-can-you-get-the-manifest-version-number-
235        //         from-the-apps-layout-xml-variable
236        final PackageManager packageManager = getPackageManager();
237        if (packageManager != null) {
238            try {
239                PackageInfo packageInfo = packageManager.getPackageInfo(getPackageName(), 0);
240                versionName = packageInfo.versionName;
241            } catch (PackageManager.NameNotFoundException e) {
242                Log.v(TAG,"failed to find versionName");
243                versionName = null;
244            }
245        }
246        tv.setText("OpenSeizureDetector Server Version "+versionName);
247
248        if (!isServerRunning()) {
249            Log.v(TAG,"Server not Running - Starting Server");
250            startServer();
251        } else {
252            Log.v(TAG,"Server Already Running OK");
253        }
254        // and bind to it so we can see its data
255        bindToServer();
256
257    }

265  /** Defines callbacks for service binding, passed to bindService() */
266    private ServiceConnection mConnection = new ServiceConnection() {
267
268        @Override
269        public void onServiceConnected(ComponentName className,
270                IBinder service) {
271            // We've bound to LocalService, cast the IBinder and get LocalService instance
272            SdServer.SdBinder binder = (SdServer.SdBinder) service;
273            mSdServer = binder.getService();
274            mBound = true;
275            if (mSdServer!=null) {
276                Log.v(TAG,"onServiceConnected() - Asking server to update its settings");
277                mSdServer.updatePrefs();
278            }
279            else {
280                Log.v(TAG,"onServiceConnected() - mSdServer is null - this is wrong!");
281            }
282        }
283
284        @Override
285        public void onServiceDisconnected(ComponentName arg0) {
286            Log.v(TAG,"onServiceDisonnected()");
287            mBound = false;
288        }
289    };

292    /**
293     * bind to an already running server.
294     */
295    private void bindToServer() {
296        Log.v(TAG,"bindToServer() - binding to SdServer");
297        Intent intent = new Intent(this,SdServer.class);
298        bindService(intent,mConnection, Context.BIND_AUTO_CREATE);
299    }
300
301    /** 
302     * unbind from server
303     */
304    private void unbindFromServer() {
305        // unbind this activity from the service if it is bound.
306        if (mBound) {
307            Log.v(TAG,"unbindFromServer() - unbinding");
308            try {
309                unbindService(mConnection);
310                mBound = false;
311            } catch (Exception ex) {
312                Log.e(TAG,"unbindFromServer() - error unbinding service - "+ex.toString());
313            }
314        } else {
315            Log.v(TAG,"unbindFromServer() - not bound to server - ignoring");
316        }
317    }

319    /**
320     * Start the SdServer service
321     */
322    private void startServer() {
323        // Start the server
324        sdServerIntent = new Intent(MainActivity.this,SdServer.class);
325        sdServerIntent.setData(Uri.parse("Start"));
326        getApplicationContext().startService(sdServerIntent);
327
328        // Change the action bar icon to show the option to stop the service.
329        if (mOptionsMenu!=null) {
330            Log.v(TAG,"Changing menu icons");
331            MenuItem menuItem = mOptionsMenu.findItem(R.id.action_start_stop);
332            menuItem.setIcon(R.drawable.stop_server);
333            menuItem.setTitle("Stop Server");
334        } else {
335            Log.v(TAG,"mOptionsMenu is null - not changing icons!");
336        }
337    }
338
339    /**
340     * Stop the SdServer service
341     */
342    private void stopServer() {
343        Log.v(TAG,"stopping Server...");
344
345        // then send an Intent to stop the service.
346        sdServerIntent = new Intent(MainActivity.this,SdServer.class);
347        sdServerIntent.setData(Uri.parse("Stop"));
348        getApplicationContext().stopService(sdServerIntent);
349
350        // Change the action bar icon to show the option to start the service.
351        if (mOptionsMenu!=null) {
352            Log.v(TAG,"Changing action bar icons");
353            mOptionsMenu.findItem(R.id.action_start_stop).setIcon(R.drawable.start_server);
354            mOptionsMenu.findItem(R.id.action_start_stop).setTitle("Start Server");
355        } else {
356            Log.v(TAG,"mOptionsMenu is null, not changing icons!");
357        }
358
359    }

364    public boolean isServerRunning() {
365        //Log.v(TAG,"isServerRunning()................");
366        ActivityManager manager = 
367            (ActivityManager) getSystemService(ACTIVITY_SERVICE);
368        for (RunningServiceInfo service : 
369                 manager.getRunningServices(Integer.MAX_VALUE)) {
370            //Log.v(TAG,"Service: "+service.service.getClassName());
371            if ("uk.org.openseizuredetector.SdServer"
372                .equals(service.service.getClassName())) {
373                //Log.v(TAG,"Yes!");
374            return true;
375            }
376        }
377        //Log.v(TAG,"No!");
378        return false;
379    }

382    /** get the ip address of the phone.
383     * Based on http://stackoverflow.com/questions/11015912/how-do-i-get-ip-address-in-ipv4-format
384     */
385    public String getLocalIpAddress() {
386        try {
387            for (Enumeration<NetworkInterface> en = NetworkInterface
388                     .getNetworkInterfaces(); en.hasMoreElements();) {
389                NetworkInterface intf = en.nextElement();
390                for (Enumeration<InetAddress> enumIpAddr = intf
391                         .getInetAddresses(); enumIpAddr.hasMoreElements();) {
392                    InetAddress inetAddress = enumIpAddr.nextElement();
393                    //Log.v(TAG,"ip1--:" + inetAddress);
394                    //Log.v(TAG,"ip2--:" + inetAddress.getHostAddress());
395                
396                    // for getting IPV4 format
397                    if (!inetAddress.isLoopbackAddress() 
398                        && InetAddressUtils.isIPv4Address(
399                                 inetAddress.getHostAddress())) {
400                    
401                        String ip = inetAddress.getHostAddress().toString();
402                        //Log.v(TAG,"ip---::" + ip);
403                        return ip;
404                    }
405                }
406            }
407        } catch (Exception ex) {
408            Log.e("IP Address", ex.toString());
409        }
410        return null;
411    }

415    /*
416     * updateServerStatus - called by the uiTimer timer periodically.
417     * requests the ui to be updated by calling serverStatusRunnable.
418     */
419    private void updateServerStatus() {
420        serverStatusHandler.post(serverStatusRunnable);
421    }
422        
423    /*
424     * serverStatusRunnable - called by updateServerStatus - updates the
425     * user interface to reflect the current status received from the server.
426     */
427    final Runnable serverStatusRunnable = new Runnable() {
428            public void run() {
429                TextView tv;
430                tv = (TextView) findViewById(R.id.textView1);
431                if (isServerRunning()) {
432                    tv.setText("Server Running OK");
433                    tv.setBackgroundColor(okColour);
434                    tv = (TextView)findViewById(R.id.textView2);
435                    tv.setText("Access Server at http://"
436                                   +getLocalIpAddress()
437                                   +":8080");
438                    tv.setBackgroundColor(okColour);
439                } else {
440                    tv.setText("*** Server Stopped ***");
441                    tv.setBackgroundColor(alarmColour);
442                }
443
444
445                try {
446                    if (mBound) {
447                        tv = (TextView) findViewById(R.id.alarmTv);
448                        if (mSdServer.sdData.alarmState==0) {
449                            tv.setText(mSdServer.sdData.alarmPhrase);
450                            tv.setBackgroundColor(okColour);
451                        }
452                        if (mSdServer.sdData.alarmState==1) {
453                            tv.setText(mSdServer.sdData.alarmPhrase);
454                            tv.setBackgroundColor(warnColour);
455                        }
456                        if (mSdServer.sdData.alarmState==2) {
457                            tv.setText(mSdServer.sdData.alarmPhrase);
458                            tv.setBackgroundColor(alarmColour);
459                        }
460                        tv = (TextView) findViewById(R.id.pebTimeTv);
461                        tv.setText(mSdServer.mPebbleStatusTime.format("%H:%M:%S"));
462                        // Pebble Connected Phrase
463                        tv = (TextView) findViewById(R.id.pebbleTv);
464                        if (mSdServer.sdData.pebbleConnected) {
465                            tv.setText("Pebble Watch Connected OK");        
466                            tv.setBackgroundColor(okColour);
467                        } else {
468                            tv.setText("** Pebble Watch NOT Connected **");
469                            tv.setBackgroundColor(alarmColour);
470                        }
471                        tv = (TextView) findViewById(R.id.appTv);
472                        if (mSdServer.sdData.pebbleAppRunning) {
473                            tv.setText("Pebble App OK");            
474                            tv.setBackgroundColor(okColour);
475                        } else {
476                            tv.setText("** Pebble App NOT Running **");     
477                            tv.setBackgroundColor(alarmColour);
478                        }
479                        tv = (TextView) findViewById(R.id.battTv);
480                        tv.setText("Pebble Battery = "+String.valueOf(mSdServer.sdData.batteryPc)+"%");
481                        if (mSdServer.sdData.batteryPc<=20)
482                            tv.setBackgroundColor(alarmColour);
483                        if (mSdServer.sdData.batteryPc>20)
484                            tv.setBackgroundColor(warnColour);
485                        if (mSdServer.sdData.batteryPc>=40)
486                            tv.setBackgroundColor(okColour);
487    
488                        tv = (TextView) findViewById(R.id.debugTv);
489                        String specStr = "";
490                        for (int i=0;i<10;i++)
491                            specStr = specStr 
492                                + mSdServer.sdData.simpleSpec[i] 
493                                + ", ";
494                        tv.setText("Spec = "+specStr);
495                    }
496                    else {
497                        tv = (TextView) findViewById(R.id.alarmTv);
498                        tv.setText("Not Connected to Server");
499                        tv.setBackgroundColor(warnColour);
500                    }
501                } catch (Exception e) {
502                    Log.v(TAG,"ServerStatusRunnable: Exception - "+e.toString());
503                }
504                ////////////////////////////////////////////////////////////
505                // Produce graph
506                LineChart mChart = (LineChart) findViewById(R.id.chart1);
507                mChart.setDescription("");
508                mChart.setNoDataTextDescription("You need to provide data for the chart.");
509                // X Values
510                ArrayList<String> xVals = new ArrayList<String>();
511                for (int i = 0; i < 10; i++) {
512                    xVals.add((i) + "");
513                }
514                // Y Values
515                ArrayList<Entry> yVals = new ArrayList<Entry>();
516                for (int i = 0; i < 10; i++) {
517                    if (mSdServer!=null)
518                        yVals.add(new Entry(mSdServer.sdData.simpleSpec[i], i));
519                    else
520                        yVals.add(new Entry(i, i));
521                }
522
523                // create a dataset and give it a type
524                LineDataSet set1 = new LineDataSet(yVals, "DataSet 1");
525                set1.setColor(Color.BLACK);
526                set1.setLineWidth(1f);
527
528                ArrayList<LineDataSet> dataSets = new ArrayList<LineDataSet>();
529                dataSets.add(set1); // add the datasets
530                LineData data = new LineData(xVals, dataSets);
531                //data.setValueTextSize(10f);
532                mChart.setData(data);
533                mChart.invalidate();
534            }
535        };
