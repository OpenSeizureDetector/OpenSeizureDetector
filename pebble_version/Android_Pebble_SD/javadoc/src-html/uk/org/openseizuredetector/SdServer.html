192    /**
193     * onStartCommand - start the web server and the message loop for
194     * communications with other processes.
195     */
196    @Override
197    public int onStartCommand(Intent intent, int flags, int startId) {
198        Log.v(TAG,"onStartCommand() - SdServer service starting");
199        
200        // Update preferences.
201        Log.v(TAG,"onStartCommand() - calling updatePrefs()");
202        updatePrefs();
203        
204        // Display a notification icon in the status bar of the phone to
205        // show the service is running.
206        Log.v(TAG,"showing Notification");
207        showNotification();
208
209        // Record last time we sent an SMS so we can limit rate of SMS
210        // sending to one per minute.
211        mSMSTime = new Time(Time.getCurrentTimezone());
212
213
214        // Start receiving data from the pebble watch
215        startPebbleServer();
216
217        // Start timer to check status of pebble regularly.
218        mPebbleStatusTime = new Time(Time.getCurrentTimezone());
219        //getPebbleStatus();
220        if (statusTimer==null) {
221            Log.v(TAG,"onCreate(): starting status timer");
222            statusTimer = new Timer();
223            statusTimer.schedule(new TimerTask() {
224                    @Override
225                    public void run() {getPebbleStatus();}
226                }, 0, 1000);    
227        } else {
228            Log.v(TAG,"onCreate(): status timer already running.");
229        }
230        
231        // Start timer to retrieve pebble settings regularly.
232        getPebbleSdSettings();
233        if (settingsTimer == null) {
234            Log.v(TAG,"onCreate(): starting settings timer");
235            settingsTimer = new Timer();
236            settingsTimer.schedule(new TimerTask() {
237                    @Override
238                    public void run() {getPebbleSdSettings();}
239                }, 0, 1000*60); 
240        } else {
241            Log.v(TAG,"onCreate(): settings timer already running.");
242        }
243
244        // Start timer to log data regularly..
245        if (dataLogTimer == null) {
246            Log.v(TAG,"onCreate(): starting dataLog timer");
247            dataLogTimer = new Timer();
248            dataLogTimer.schedule(new TimerTask() {
249                    @Override
250                    public void run() {logData();}
251                }, 0, 1000*60); 
252        } else {
253            Log.v(TAG,"onCreate(): dataLog timer already running.");
254        }
255
256
257        // Start the web server
258        startWebServer();
259
260        // Apply the wake-lock to prevent CPU sleeping (very battery intensive!)
261        if (mWakeLock!=null) {
262            mWakeLock.acquire();
263            Log.v(TAG,"Applied Wake Lock to prevent device sleeping");
264        } else {
265            Log.d(TAG,"mmm...mWakeLock is null, so not aquiring lock.  This shouldn't happen!");
266        }
267
268        return START_STICKY;
269    }

271    @Override
272    public void onDestroy() {
273        Log.v(TAG,"onDestroy(): SdServer Service stopping");
274        // release the wake lock to allow CPU to sleep and reduce
275        // battery drain.
276        if (mWakeLock!=null) {
277            mWakeLock.release();
278            Log.v(TAG,"Released Wake Lock to allow device to sleep.");
279        } else {
280            Log.d(TAG,"mmm...mWakeLock is null, so not releasing lock.  This shouldn't happen!");
281        }
282
283        try {
284            // Stop the status timer
285            if (statusTimer!=null) {
286                Log.v(TAG,"onDestroy(): cancelling status timer");
287                statusTimer.cancel();
288                statusTimer.purge();
289                statusTimer = null;
290            }
291            // Stop the settings timer
292            if (settingsTimer!=null) {
293                Log.v(TAG,"onDestroy(): cancelling settings timer");
294                settingsTimer.cancel();
295                settingsTimer.purge();
296                settingsTimer = null;
297            }
298            // Cancel the notification.
299            Log.v(TAG,"onDestroy(): cancelling notification");
300            mNM.cancel(NOTIFICATION_ID);
301            // Stop web server
302            Log.v(TAG,"onDestroy(): stopping web server");
303            stopWebServer();
304            // Stop pebble message handler.
305            Log.v(TAG,"onDestroy(): stopping pebble server");
306            stopPebbleServer();
307            // stop this service.
308            Log.v(TAG,"onDestroy(): calling stopSelf()");
309            stopSelf();
310
311        } catch(Exception e) {
312            Log.v(TAG,"Error in onDestroy() - "+e.toString());
313        }
314    }

421    /**
422     * Set this server to receive pebble data by registering it as
423     * A PebbleDataReceiver
424     */
425    private void startPebbleServer() {
426        Log.v(TAG,"StartPebbleServer()");
427        final Handler handler = new Handler();
428        msgDataHandler = new PebbleKit.PebbleDataReceiver(SD_UUID) {
429                @Override
430                public void receiveData(final Context context,
431                                        final int transactionId,
432                                        final PebbleDictionary data) {
433                    Log.v(TAG,"Received message from Pebble - data type="
434                          +data.getUnsignedIntegerAsLong(KEY_DATA_TYPE));
435                    // If we have a message, the app must be running
436                    mPebbleAppRunningCheck = true;  
437                    PebbleKit.sendAckToPebble(context,transactionId);
438                    //Log.v(TAG,"Message is: "+data.toJsonString());
439                    if (data.getUnsignedIntegerAsLong(KEY_DATA_TYPE)
440                        ==DATA_TYPE_RESULTS) {
441                        Log.v(TAG,"DATA_TYPE = Results");
442                        sdData.dataTime.setToNow();
443                        Log.v(TAG,"sdData.dataTime="+sdData.dataTime);
444
445                        sdData.alarmState = data.getUnsignedIntegerAsLong(
446                                                                   KEY_ALARMSTATE);
447                        sdData.maxVal = data.getUnsignedIntegerAsLong(KEY_MAXVAL);
448                        sdData.maxFreq = data.getUnsignedIntegerAsLong(KEY_MAXFREQ);
449                        sdData.specPower = data.getUnsignedIntegerAsLong(KEY_SPECPOWER);
450                        sdData.roiPower = data.getUnsignedIntegerAsLong(KEY_ROIPOWER);
451                        sdData.alarmPhrase = "Unknown";
452                        if (sdData.alarmState==0) {
453                            sdData.alarmPhrase="OK";
454                        }
455                        if (sdData.alarmState==1) {
456                            sdData.alarmPhrase="WARNING";
457                            if (mLogAlarms) {
458                                Log.v(TAG,"WARNING - Loggin to SD Card");
459                                writeAlarmToSD();
460                                logData();
461                            } else {
462                                Log.v(TAG,"WARNING");
463                            }
464                            warningBeep();
465                        }
466                        if (sdData.alarmState==2) {
467                            sdData.alarmPhrase="ALARM";
468                            if (mLogAlarms) {
469                                Log.v(TAG,"***ALARM*** - Loggin to SD Card");
470                                writeAlarmToSD();
471                                logData();
472                            } else {
473                                Log.v(TAG,"***ALARM***");
474                            }
475                            // Make alarm beep tone
476                            alarmBeep();
477                            // Send SMS Alarm.
478                            if (mSMSAlarm) {
479                                Time tnow = new Time(Time.getCurrentTimezone());
480                                tnow.setToNow();
481                                // limit SMS alarms to one per minute
482                                if  ((tnow.toMillis(false) 
483                                      - mSMSTime.toMillis(false)) 
484                                     > 60000) {
485                                sendSMSAlarm();
486                                mSMSTime = tnow;
487                                }
488                            }
489                        }
490
491                        // Read the data that has been sent, and convert it into
492                        // an integer array.
493                        byte[] byteArr = data.getBytes(KEY_SPEC_DATA);
494                        IntBuffer intBuf = ByteBuffer.wrap(byteArr)
495                            .order(ByteOrder.LITTLE_ENDIAN)
496                            .asIntBuffer();
497                        int[] intArray = new int[intBuf.remaining()];
498                        intBuf.get(intArray);
499                        for (int i=0;i<intArray.length;i++) {
500                            sdData.simpleSpec[i] = intArray[i];
501                        }
502
503
504                    }
505                    if (data.getUnsignedIntegerAsLong(KEY_DATA_TYPE)
506                        ==DATA_TYPE_SETTINGS) {
507                        Log.v(TAG,"DATA_TYPE = Settings");
508                        sdData.alarmFreqMin = data.getUnsignedIntegerAsLong(KEY_ALARM_FREQ_MIN);
509                        sdData.alarmFreqMax = data.getUnsignedIntegerAsLong(KEY_ALARM_FREQ_MAX);
510                        sdData.nMin = data.getUnsignedIntegerAsLong(KEY_NMIN);
511                        sdData.nMax = data.getUnsignedIntegerAsLong(KEY_NMAX);
512                        sdData.warnTime = data.getUnsignedIntegerAsLong(KEY_WARN_TIME);
513                        sdData.alarmTime = data.getUnsignedIntegerAsLong(KEY_ALARM_TIME);
514                        sdData.alarmThresh = data.getUnsignedIntegerAsLong(KEY_ALARM_THRESH);
515                        sdData.alarmRatioThresh = data.getUnsignedIntegerAsLong(KEY_ALARM_RATIO_THRESH);
516                        sdData.batteryPc = data.getUnsignedIntegerAsLong(KEY_BATTERY_PC);
517                        sdData.haveSettings = true;
518                    }   
519                }
520            };
521        PebbleKit.registerReceivedDataHandler(this,msgDataHandler);
522    }

598    /** 
599     * Checks the status of the connection to the pebble watch,
600     * and sets class variables for use by other functions.
601     * If the watch app is not running, it attempts to re-start it.
602     */
603    public void getPebbleStatus() {
604        Time tnow = new Time(Time.getCurrentTimezone());
605        tnow.setToNow();
606        // Check we are actually connected to the pebble.
607        sdData.pebbleConnected = PebbleKit.isWatchConnected(this);
608        // And is the pebble_sd app running?
609        // set mPebbleAppRunningCheck has been false for more than 10 seconds
610        // the app is not talking to us
611        // mPebbleAppRunningCheck is set to true in the receiveData handler. 
612        if (!mPebbleAppRunningCheck && 
613            ((tnow.toMillis(false) - mPebbleStatusTime.toMillis(false)) > 10000)) {
614            Log.v(TAG,"tdiff = "+(tnow.toMillis(false) - mPebbleStatusTime.toMillis(false)));
615            sdData.pebbleAppRunning = false;
616            Log.v(TAG,"Pebble App Not Running - Attempting to Re-Start");
617            startWatchApp();
618            getPebbleSdSettings();
619            if (mAudibleFaultWarning) {
620                faultWarningBeep();
621            }
622        } else {
623            sdData.pebbleAppRunning = true;
624        }
625
626        // if we have confirmation that the app is running, reset the
627        // status time to now and initiate another check.
628        if (mPebbleAppRunningCheck) {
629            mPebbleAppRunningCheck = false;
630            mPebbleStatusTime.setToNow();
631        }
632
633        if (!sdData.haveSettings) {
634            Log.v(TAG,"getPebbleStatus() - no settings received yet - requesting");
635            getPebbleSdSettings();
636        }
637    }

654    /**
655     * updatePrefs() - update basic settings from the SharedPreferences
656     * - defined in res/xml/prefs.xml
657     */
658    public void updatePrefs() {
659        Log.v(TAG,"updatePrefs()");
660        SharedPreferences SP = PreferenceManager
661            .getDefaultSharedPreferences(getBaseContext());
662        try {
663            mAudibleFaultWarning = SP.getBoolean("AudibleFaultWarning",true);
664            Log.v(TAG,"updatePrefs() - mAuidbleFaultWarning = "+mAudibleFaultWarning);
665            mAudibleAlarm = SP.getBoolean("AudibleAlarm",true);
666            Log.v(TAG,"updatePrefs() - mAuidbleAlarm = "+mAudibleAlarm);
667            mAudibleWarning = SP.getBoolean("AudibleWarning",true);
668            Log.v(TAG,"updatePrefs() - mAuidbleWarning = "+mAudibleWarning);

761     * in which case writes to alarm log file.
762     */
763    public void writeToSD(boolean alarm) {
764        Log.v(TAG,"writeToSD("+alarm+")");
765        Time tnow = new Time(Time.getCurrentTimezone());
766        tnow.setToNow();
767        String dateStr = tnow.format("%Y-%m-%d");
768
769        // Select filename depending on 'alarm' parameter.
770        String fname;
771        if (alarm) 
772            fname = "AlarmLog";
773        else
774            fname = "DataLog";
775
776        fname = fname+"_"+dateStr+".txt";
777        // Open output directory on SD Card.
778        if (isExternalStorageWritable()) {
779            try {
780                FileWriter of = new FileWriter(getDataStorageDir().toString() 
781                                               + "/" + fname, true);
782                if (sdData!=null) {
783                    Log.v(TAG,"writing sdData.toString()");
784                    of.append(sdData.toString()+"\n");
785                }
786                of.close();
787            } catch(Exception ex) {
788                Log.e(TAG,"writeAlarmToSD - error "+ex.toString());
789            }    
790        } else {
791            Log.e(TAG,"ERROR - Can not Write to External Folder");
792        }

913        /**
914         * Return a file from the external storage folder
915         */
916        NanoHTTPD.Response serveLogFile(String uri) {
917            NanoHTTPD.Response res;
918            InputStream ip = null;
919            String uripart;
920            Log.v(TAG,"serveLogFile("+uri+")");
921            try {
922                if (ip!=null) ip.close();
923                String storageDir = getDataStorageDir().toString();
924                StringTokenizer uriParts = new StringTokenizer(uri,"/");
925                Log.v(TAG,"serveExternalFile - number of tokens = "+uriParts.countTokens());        
926                while (uriParts.hasMoreTokens()) {
927                    uripart = uriParts.nextToken();
928                    Log.v(TAG,"uripart="+uripart);
929                }
930
931                // If we have only given a "/logs" URI, return a list of
932                // available files.
933                // Re-start the StringTokenizer from the start.
934                uriParts = new StringTokenizer(uri,"/");
935                Log.v(TAG,"serveExternalFile - number of tokens = "
936                      +uriParts.countTokens());     
937                if (uriParts.countTokens()==1) {
938                    Log.v(TAG,"Returning list of files");
939                    
940                    File dirs = getDataStorageDir();
941                    try {
942                        JSONObject jsonObj = new JSONObject();
943                        if (dirs.exists()) {
944                            String[] fileList = dirs.list();
945                            JSONArray arr = new JSONArray();
946                            for (int i=0;i<fileList.length;i++)
947                                arr.put(fileList[i]);
948                            jsonObj.put("logFileList",arr);
949                        }
950                        res = new NanoHTTPD.Response(NanoHTTPD.Response.Status.OK,
951                                                     "text/html",jsonObj.toString());
952                    } catch(Exception ex) {
953                        res = new NanoHTTPD.Response(NanoHTTPD.Response.Status.OK,
954                                                     "text/html","ERROR - "+ex.toString());
955                    }
956                    return res;
957                }
958
959                uripart = uriParts.nextToken();  // This will just be /logs
960                uripart = uriParts.nextToken();  // this is the requested file.
961                String fname = storageDir + "/"+uripart;
962                Log.v(TAG,"serveLogFile - uri="+uri+", fname="+fname);
963                ip = new FileInputStream(fname);
964                String mimeStr = "text/html";
965                res = new NanoHTTPD.Response(NanoHTTPD.Response.Status.OK,
966                                             mimeStr,ip);
967                res.addHeader("Content-Length", "" + ip.available());
968            } catch (IOException ex) {
969                Log.v(TAG,"serveLogFile(): Error Opening File - "+ex.toString());
970                res = new NanoHTTPD.Response("serveLogFile(): Error Opening file "+uri);
971            } 
972            return(res);
973        }
